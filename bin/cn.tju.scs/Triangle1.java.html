package cn.tju.scs;

public class Triangle1 {
	public boolean istriangle(int a,int b,int c){
		 if(a+b>c&&a+c>b&&b+c>a){
		     return true;
		     }
		 else
		     return false;
		}

		public int form(int a ,int b, int c){
		       if(istriangle(a,b,c)){
		           if(a==b&&b==c){
		            return 3;
		            }
		           else if((a==b)||(a==c)||(b==c)){         
		             return 2;
		}
		         else
		          return 1;
		}
		else 
		           return 0;
		}

}
